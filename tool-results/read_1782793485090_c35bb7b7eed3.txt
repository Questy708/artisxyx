   110→   110→/* ── Relative Time Formatter ── */
   111→   111→function formatRelativeTime(dateStr: string): string {
   112→   112→  try {
   113→   113→    const date = new Date(dateStr);
   114→   114→    const now = new Date();
   115→   115→    const diffMs = now.getTime() - date.getTime();
   116→   116→    const diffSec = Math.floor(diffMs / 1000);
   117→   117→    const diffMin = Math.floor(diffSec / 60);
   118→   118→    const diffHr = Math.floor(diffMin / 60);
   119→   119→    const diffDay = Math.floor(diffHr / 24);
   120→   120→
   121→   121→    if (diffSec < 60) return "Just now";
   122→   122→    if (diffMin < 60) return `${diffMin} min. ago`;
   123→   123→    if (diffHr < 24) return `${diffHr} hr. ago`;
   124→   124→    if (diffDay === 1) return "1 day ago";
   125→   125→    if (diffDay < 30) return `${diffDay} days ago`;
   126→   126→    return date.toLocaleDateString();
   127→   127→  } catch {
   128→   128→    return dateStr;
   129→   129→  }
   130→   130→}
