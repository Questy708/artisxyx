   110→/* ── Relative Time Formatter ── */
   111→function formatRelativeTime(dateStr: string): string {
   112→  try {
   113→    const date = new Date(dateStr);
   114→    const now = new Date();
   115→    const diffMs = now.getTime() - date.getTime();
   116→    const diffSec = Math.floor(diffMs / 1000);
   117→    const diffMin = Math.floor(diffSec / 60);
   118→    const diffHr = Math.floor(diffMin / 60);
   119→    const diffDay = Math.floor(diffHr / 24);
   120→
   121→    if (diffSec < 60) return "Just now";
   122→    if (diffMin < 60) return `${diffMin} min. ago`;
   123→    if (diffHr < 24) return `${diffHr} hr. ago`;
   124→    if (diffDay === 1) return "1 day ago";
   125→    if (diffDay < 30) return `${diffDay} days ago`;
   126→    return date.toLocaleDateString();
   127→  } catch {
   128→    return dateStr;
   129→  }
   130→}

   132→/* ── Recently Active Helper ── */
   133→function isRecentlyActive(dateStr: string): boolean {
   134→  try {
   135→    const date = new Date(dateStr);
   136→    const now = new Date();
   137→    const diffMs = now.getTime() - date.getTime();
   138→    const diffHr = diffMs / (1000 * 60 * 60);
   139→    return diffHr <= 12;
   140→  } catch {
   141→    return false;
   142→  }
   143→}

   320→function OnboardingFlow({ onComplete }: { onComplete: (user: ForumUser) => void }) {
   321→  const [step, setStep] = useState(0);
   322→  const [direction, setDirection] = useState(1);
   323→  const [displayName, setDisplayName] = useState("");
   324→  const [email, setEmail] = useState("");
   325→  const [bio, setBio] = useState("");
   326→  const [role, setRole] = useState("");
   327→  const [location, setLocation] = useState("");
   328→  const [selectedCommunities, setSelectedCommunities] = useState<string[]>([]);
   329→  const [isSubmitting, setIsSubmitting] = useState(false);
   330→  const [avatarUrl, setAvatarUrl] = useState<string>("");
   331→  const [avatarFile, setAvatarFile] = useState<File | null>(null);
   332→
   333→  const goNext = () => { setDirection(1); setStep((s) => s + 1); };
   334→  const goBack = () => { setDirection(-1); setStep((s) => s - 1); };
   335→
   336→  const toggleCommunity = (name: string) => {
   337→    setSelectedCommunities((prev) =>
   338→      prev.includes(name) ? prev.filter((c) => c !== name) : [...prev, name]
   339→    );
   340→  };
   341→
   342→  const handleComplete = async () => {
   343→    setIsSubmitting(true);
   344→    try {
   345→      const avatarColor = ROLE_COLORS[role] || "#6b7280";
   346→      let finalAvatarUrl = avatarUrl;
   347→
   348→      // Upload avatar file if selected (skip gracefully if /api/upload doesn't exist)
   349→      if (avatarFile) {
   350→        try {
   351→          const formData = new FormData();
   352→          formData.append("file", avatarFile);
   353→          const uploadRes = await fetch("/api/upload", {
   354→            method: "POST",
   355→            body: formData,
   356→          });
   357→          if (uploadRes.ok) {
   358→            const uploadData = await uploadRes.json();
   359→            finalAvatarUrl = uploadData.url;
   360→          }
   361→        } catch (uploadErr) {
   362→          console.warn("Avatar upload skipped:", uploadErr);
   363→        }
   364→      }
   365→
   366→      // Create user via API
   367→      const res = await fetch("/api/forum/users", {
   368→        method: "POST",
   369→        headers: { "Content-Type": "application/json" },
   370→        body: JSON.stringify({
   371→          name: displayName,
   372→          email,
   373→          bio: bio || undefined,
   374→          role,
   375→          location: location || undefined,
   376→          communities: selectedCommunities,
   377→          avatarColor,
   378→          avatarUrl: finalAvatarUrl || undefined,
   379→        }),
   380→      });
   381→
   382→      if (!res.ok) {
   383→        let err;
   384→        try { err = await res.json(); } catch { err = { error: `Server returned ${res.status}` }; }
   385→        console.error("Failed to create user:", err);
   386→        setIsSubmitting(false);
   387→        return;
   388→      }
   389→
   390→      const user: ForumUser = await res.json();
   391→
   392→      // Store user ID in localStorage
   393→      localStorage.setItem("xcelero_townsquare_user_id", user.id);
   394→
   395→      // Seed the database with sample data (use public init route, not admin-gated seed)
   396→      await fetch("/api/forum/init").catch(() => {});
   397→
   398→      onComplete(user);
   399→    } catch (err) {
   400→      console.error("Onboarding error:", err);
   401→      setIsSubmitting(false);
   402→    }
   403→  };
   404→
   405→  return (
   406→    <div className="flex flex-col min-h-screen bg-white text-[#111111] font-sans">
   407→      {/* Header bar */}
   408→      <header className="h-[56px] bg-white border-b border-[#111111]/10 px-4 md:px-8 flex items-center justify-between shrink-0">
   409→        <div className="flex items-center gap-2.5">
   410→          <div className="w-8 h-8 bg-[#FF4D00] flex items-center justify-center">
   411→            <MessageSquare className="w-4 h-4 text-white" />
   412→          </div>
   413→          <span className="text-lg font-display font-medium tracking-tight text-[#111111]">
   414→            Town<span className="text-[#111111]/30 font-normal"> Square</span>
   415→          </span>
   416→        </div>
   417→        {/* Step indicator */}
   418→        <div className="flex items-center gap-2">
   419→          {[0, 1, 2, 3].map((i) => (
   420→            <div
   421→              key={i}
   422→              className={`h-1 rounded-full transition-all duration-300 ${
   423→                i === step ? "w-8 bg-[#FF4D00]" : i < step ? "w-4 bg-[#FF4D00]/40" : "w-4 bg-[#111111]/10"
   424→              }`}
   425→            />
   426→          ))}
   427→        </div>

   429→      </header>
