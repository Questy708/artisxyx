   439→/* ══════════════════════════════════════════════════════════════════════════
   440→   NUMBERS SECTION: Animated counting stats
   441→   ══════════════════════════════════════════════════════════════════════════ */
   442→function AnimatedCounter({ value, prefix, suffix, duration = 2 }: {
   443→  value: number;
   444→  prefix: string;
   445→  suffix: string;
   446→  duration?: number;
   447→}) {
   448→  const [count, setCount] = useState(0);
   449→  const [hasAnimated, setHasAnimated] = useState(false);
   450→  const ref = useRef<HTMLDivElement>(null);
   451→
   452→  useEffect(() => {
   453→    const el = ref.current;
   454→    if (!el || hasAnimated) return;
   455→
   456→    const observer = new IntersectionObserver(
   457→      ([entry]) => {
   458→        if (entry.isIntersecting && !hasAnimated) {
   459→          setHasAnimated(true);
   460→          observer.disconnect();
   461→
   462→          let startTime: number | null = null;
   463→          let animationFrame: number;
   464→
   465→          const animate = (timestamp: number) => {
   466→            if (!startTime) startTime = timestamp;
   467→            const progress = Math.min((timestamp - startTime) / (duration * 1000), 1);
   468→            const eased = 1 - Math.pow(1 - progress, 3);
   469→            setCount(value * eased);
   470→
   471→            if (progress < 1) {
   472→              animationFrame = requestAnimationFrame(animate);
   473→            }
   474→          };
   475→
   476→          animationFrame = requestAnimationFrame(animate);
   477→          return () => cancelAnimationFrame(animationFrame);
   478→        }
   479→      },
   480→      { threshold: 0.2 }
   481→    );
   482→
   483→    observer.observe(el);
   484→    return () => observer.disconnect();
   485→  }, [hasAnimated, value, duration]);
   486→
   487→  const displayValue = value % 1 === 0 ? Math.round(count) : count.toFixed(1);
   488→
   489→  return (
   490→    <div ref={ref} className="tabular-nums">
   491→      <span className="text-[36px] sm:text-[48px] md:text-[72px] lg:text-[88px] font-display font-medium tracking-[-0.03em] leading-[1]">
   492→        {prefix}{displayValue}{suffix}
   493→      </span>
   494→    </div>
   495→  );
   496→}
